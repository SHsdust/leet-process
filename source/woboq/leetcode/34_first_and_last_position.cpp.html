10 while(l < r){

13 r = mid;

32 vector<int> searchRange(vector<int>& nums, int target) {
33 int n(nums.size());
34 if(!n) return vector<int>({-1, -1});
35
36 int lower(lower_bound(nums, target));
37 int upper(upper_bound(nums, target) - 1);
38
39 if(lower == n || nums[lower] != target)
40 return vector<int>({-1, -1});
41
42 return vector<int>({lower, upper});
43
44 }
