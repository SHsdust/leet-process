4class Solution {
5public:
6 void backtracking(vector<vector<string>> &ans, vector<string> &board, \
7 vector<bool> &column, vector<bool> &ldiag, vector<bool> &rdiag, int row, int n) {
8 if (row == n) {
9 ans.push_back(board);
10 return;
11 }
12 for (int i = 0; i < n; ++i) {
13 if (column[i] || ldiag[n-row+i-1] || rdiag[row+i+1]) {
14 continue;
15 }
16 board[row][i] = 'Q';
17 column[i] = ldiag[n-row+i-1] = rdiag[row+i+1] = true;
18 backtracking(ans, board, column, ldiag, rdiag, row+1, n);
19 board[row][i] = '.';
20 column[i] = ldiag[n-row+i-1] = rdiag[row+i+1] = false;
21 }
22 }
23 vector<vector<string>> solveNQueens(int n) {
24 vector<vector<string>> ans;
25 if (n == 0) {
26 return ans;
27 }
28 vector<string> board(n, string(n, '.'));
29 vector<bool> column(n, false), ldiag(2*n-1, false), rdiag(2*n-1, false);
30 backtracking(ans, board, column, ldiag, rdiag, 0, n);
31 return ans;
32 }
33};
