10 vector<vector<int>> pacificAtlantic(vector<vector<int>>& heights) {
11 if(heights.empty() || heights[0].empty())
12 return {};
13
14 vector<vector<int>> ans;
15 int m(heights.size()), n(heights[0].size());
16 vector<vector<bool>> can_reach_p(m, vector<bool>(n, false));
17 vector<vector<bool>> can_reach_a(m, vector<bool>(n, false));
18
19 for(int i=0; i<m; ++i){
20 dfs(heights, can_reach_p, i, 0);
21 dfs(heights, can_reach_a, i, n-1);
22 }
23 for(int i=0; i<n; ++i){
24 dfs(heights, can_reach_p, 0, i);
25 dfs(heights, can_reach_a, m-1, i);
26 }
27 for(int i=0; i<m; ++i)
28 for(int j=0; j<n; ++j)
29 if(can_reach_p[i][j] && can_reach_a[i][j])
30 ans.push_back(vector<int>{i,j});
31 return ans;
32 }
33
34 // pass by value could cause TLE
35 void dfs(const vector<vector<int>> &heights, vector<vector<bool>> &can_reach, int r, int c){
36 if(can_reach[r][c]) return ;
37 can_reach[r][c] = true;
38 int x, y ;
39 for(int i=0; i<4; ++i){
40 x = r + direction[i];
41 y = c + direction[i+1];
42 if(x>=0 && x<heights.size() && y>=0 && y<heights[0].size() \
43 && heights[r][c]<=heights[x][y])
44 dfs(heights, can_reach, x, y);
45 }
46 }
47};
