44class Solution {
45public:
46 int maxAreaOfIsland(vector<vector<int>>& grid) {
47 if(grid.size()==0 || grid[0].size()==0) return 0;
48 int max_area(0);
49 for(int i=0; i<grid[0].size(); ++i){
50 for(int j=0; j<grid.size(); ++j){
51 if(grid[j][i]==1){
52 max_area = max(max_area, dfs(grid, i, j));
53 }
54 }
55 }
56 return max_area;
57 }
58
59
60 int dfs(vector<vector<int>> &grid, int i, int j){
61 if(grid[j][i]==0) return 0;
62 grid[j][i] = 0;
63 int area = 1;
64 int x, y;
65 for(int k=0; k<4; ++k){
66 x = i + directions[k];
67 y = j + directions[k+1];
68 if(x>=0 && x<grid[0].size() && y>=0 && y<grid.size()){
69 area += dfs(grid, x, y);
70 }
71 }
72 return area;
73 }
74};
