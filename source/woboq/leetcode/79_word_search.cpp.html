4class Solution {
5public:
6 void backtracking(int i, int j, vector<vector<char>>& board, string& word, bool
7 & find, vector<vector<bool>>& visited, int pos) {
8 if (i < 0 || i >= board.size() || j < 0 || j >= board[0].size()) {
9 return;
10 }
11 if (visited[i][j] || find || board[i][j] != word[pos]) {
12 return;
13 }
14 if (pos == word.size() - 1) {
15 find = true;
16 return;
17 }
18
19 visited[i][j] = true;
20 backtracking(i + 1, j, board, word, find, visited, pos + 1);
21 backtracking(i - 1, j, board, word, find, visited, pos + 1);
22 backtracking(i, j + 1, board, word, find, visited, pos + 1);
23 backtracking(i, j - 1, board, word, find, visited, pos + 1);
24 visited[i][j] = false;
25 }
26
27 bool exist(vector<vector<char>>& board, string word) {
28 if (board.empty()) return false;
29 int m = board.size(), n = board[0].size();
30 vector<vector<bool>> visited(m, vector<bool>(n, false));
31 bool find = false;
32 for (int i = 0; i < m; ++i) {
33 for (int j = 0; j < n; ++j) {
34 backtracking(i, j, board, word, find, visited, 0);
35 }
36 }
37 return find;
38 }

40};
