10 int l(0), r(nums.size()-1), mid;
11 while(l <= r){
12 mid = l + (r - l)/2;
13 if(nums[mid] == target)
14 return true;
15 if(nums[mid] == nums[l])
16 ++l;
17 else if(nums[mid] <= nums[r]){
18 if(target > nums[mid] && target <= nums[r])
19 l = mid +1;
20 else
21 r = mid - 1;
22 }
23 else{
24 if(target >= nums[l] && target < nums[mid])
25 r = mid - 1;
26 else
27 l = mid + 1;
28 }
29 }
30 return false;
31 }
