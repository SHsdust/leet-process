8class Solution {
9public:
10 string frequencySort(string s) {
11 unordered_map<char, int> freq;
12 int max_count(0);
13
14 for(const char c : s)
15 max_count = max(max_count, ++freq[c]);
16
17 vector<vector<int>> buckets(max_count + 1);
18 for(const auto p : freq)
19 buckets[p.second].push_back(p.first);
20
21 string ans;
22 for(int i=max_count; i>=0; --i)
23 for(const auto p : buckets[i]){
24 for(int j=0; j<i; ++j)
25 ans.push_back(p);
26 }
27
28 return ans;
29 }
30};
