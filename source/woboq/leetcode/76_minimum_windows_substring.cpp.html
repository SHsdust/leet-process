23 if(flags[s[r]])
24 if(--chars[s[r]] >= 0)
25 ++cnt;
26 while(cnt == m){
27 if(r-l+1 < min_size){
28 min_l = l;
29 min_size = r - l + 1;
30 }
31 if(flags[s[l]] && ++chars[s[l]]>0)
32 --cnt;
33 ++l;
34 }
35 }
36 if(min_size > n)
37 return "";
38 return s.substr(min_l, min_size);
39 }
