10 int res(0), pos(0), si(ratings.size()), pos2(ratings.size()-1);
11 int *assign = new int[si];
12 for(int i=0; i<si; ++i)
13 assign[i] = 1;
14
15 for(; pos < si-1; ++pos)
16 if(ratings[pos] < ratings[pos+1])
17 assign[pos+1] = assign[pos] + 1;
18
19 for(; pos2>0; --pos2)
20 if(ratings[pos2] < ratings[pos2-1])
21 if(assign[pos2] >= assign[pos2-1])
22 assign[pos2-1] = assign[pos2] + 1;
23
24 for(int i=0; i<si; ++i)
25 res += assign[i];
26 return res;
27 }
