7class Solution {
8public:
9 vector<int> direction{-1, 0, 1, 0, -1};
10
11 void dfs(queue<pair<int, int>>& points, vector<vector<int>>& grid, int m, int n
12 , int i, int j) {
13 if (i < 0 || j < 0 || i == m || j == n || grid[i][j] == 2) {
14 return;
15 }
16 if (grid[i][j] == 0) {
17 points.push({i, j});
18 return;
19 }
20 grid[i][j] = 2;
21 dfs(points, grid, m, n, i - 1, j);
22 dfs(points, grid, m, n, i + 1, j);
23 dfs(points, grid, m, n, i, j - 1);
24 dfs(points, grid, m, n, i, j + 1);
25 }
26
27 int shortestBridge(vector<vector<int>>& grid) {
28 int m = grid.size(), n = grid[0].size();
29 queue<pair<int, int>> points;
30 bool flipped = false;
31 for (int i = 0; i < m; ++i) {
32 if (flipped) break;
33 for (int j = 0; j < n; ++j) {
34 if (grid[i][j] == 1) {
35 dfs(points, grid, m, n, i, j);
36 flipped = true;
37 break;
38 }
39 }
40 }
41 int x, y;
42 int level = 0;
43 while (!points.empty()){
44 ++level;
45 int n_points = points.size();
46 while (n_points--) {
47 auto [r, c] = points.front();
48 points.pop();
49 for (int k = 0; k < 4; ++k) {
50 x = r + direction[k], y = c + direction[k+1];
51 if (x >= 0 && y >= 0 && x < m && y < n) {
52 if (grid[x][y] == 2) {
53 continue;
54 }
55 if (grid[x][y] == 1) {
56 return level;
57 }
58 points.push({x, y});
59 grid[x][y] = 2;
60 }
61 }
62 }
63 }
64 return 0;
65 }
