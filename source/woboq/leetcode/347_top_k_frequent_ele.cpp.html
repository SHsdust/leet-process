9 unordered_map<int ,int> freq;

11 for(const int num : nums)
12 max_count = max(max_count, ++freq[num]);

14 vector<vector<int>> buckets(max_count+1);
15 for(const auto p : freq)
16 buckets[p.second].push_back(p.first);
17
18 vector<int> ans;
19 for(int i=max_count; i>=0 && ans.size()<k; --i)
20 for(const int p : buckets[i]){
21 ans.push_back(p);
22 if(ans.size()==k)
23 break;
24 }
25 return ans;
26 }
