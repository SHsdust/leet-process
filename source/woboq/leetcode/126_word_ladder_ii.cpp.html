8class Solution {

10 void backtracking(const string &src, const string &dst, unordered_map<string,
11 vector<string>> &next, vector<string> &path, vector<vector<string>> &ans) {
12 if (src == dst) {
13 ans.push_back(path);
14 return;
15 }
16 for (const auto &s: next[src]) {
17 path.push_back(s);
18 backtracking(s, dst, next, path, ans);
19 path.pop_back();
20 }
21 }

23 vector<vector<string>> findLadders(string beginWord, string endWord, vector<
24 string>& wordList) {
25 vector<vector<string>> ans;
26 unordered_set<string> dict;
27 for (const auto &w: wordList){
28 dict.insert(w);
29 }
30 if (!dict.count(endWord)) {
31 return ans;
32 }
33 dict.erase(beginWord);
34 dict.erase(endWord);
35 unordered_set<string> q1{beginWord}, q2{endWord};
36 unordered_map<string, vector<string>> next;
37 bool reversed = false, found = false;
38 while (!q1.empty()) {
39 unordered_set<string> q;
40 for (const auto &w: q1) {
41 string s = w;
42 for (size_t i = 0; i < s.size(); i++) {
43 char ch = s[i];
44 for (int j = 0; j < 26; j++) {
45 s[i] = j + 'a';
46 if (q2.count(s)) {
47 reversed? next[s].push_back(w): next[w].push_back(s);
48 found = true;
49 }
50 if (dict.count(s)) {
51 reversed? next[s].push_back(w): next[w].push_back(s);
52 q.insert(s);
53 }
54 }
55 s[i] = ch;
56 }
57 }
58 if (found) {
59 break;
60 }
61 for (const auto &w: q) {
62 dict.erase(w);
63 }
64 if (q.size() <= q2.size()) {
65 q1 = q;
66 } else {
67 reversed = !reversed;
68 q1 = q2;
69 q2 = q;
70 }
71 }
72 if (found) {
73 vector<string> path = {beginWord};
74 backtracking(beginWord, endWord, next, path, ans);
75 }
76 return ans;
77 }
78};
