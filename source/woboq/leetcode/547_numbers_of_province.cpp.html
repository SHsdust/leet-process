12 if(!visited[i]){
13 ++count;
14 dfs(isConnected, visited, i);
15 }

20 void dfs(vector<vector<int>>& isConnected, vector<bool> &visited, int i){
21 visited[i] = true;
22 for(int j=0; j<isConnected.size(); ++j){
23 if(isConnected[i][j]==1 && !visited[j]){
24 visited[j] = true;
25 dfs(isConnected, visited, j);
26 }
27 }
28 }
29};
